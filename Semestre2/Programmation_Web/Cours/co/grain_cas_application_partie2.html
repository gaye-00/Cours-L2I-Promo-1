Dynamisation du cas d'application avec JavaScript et les formulaires

Introduction :

Dans cette activité, nous proposons d'étendre notre site « monSite » en y ajoutant une page contenant un formulaire qui permet à des potentiels recruteurs de nous contacter, en renseignant quelques informations en rapport avec nos critères de recrutement. Nous allons renverser les conditions de recrutement : au lieu d'aller vers les recruteurs (par ce que nous avons besoin d'eux), c'est à eux de venir vers nous (par ce que c'est eux qui ont besoin de nous).

Pour cela, nous allons :

  • Ajouter une page (appelé "recrutement.html") contenant le formulaire de recrutement

  • mettre en place un menu englobant les différents liens vers les différentes pages de notre site.

Activité 2.1 : Création de la page « recrutement.html »

Il faut savoir que toutes les pages de notre site vont avoir la même structure, c'est-à-dire, elles comporteront les mêmes grandes parties : partie gauche (les informations d'état-civile + photo + le menu du site en bas), commune à toute les pages ; la partie droite (le header, commune à toutes les pages ; le reste du contenu de la partie droite, propre à chaque page).

Ainsi, nous pouvons créer la page « rectrutement.html » à la racine du site (au même niveau que "index.html" et copier et y coller tout le contenu de la page « index.html ».

Ensuite, il faudra modifier le contenu de la balise « title » en mettant le texte « Recrutez-moi ! » et effacer toute la partie entre la balise fermante « </header> » et la balise « </div> » fermant la partie droite de la page (<div class="droite">) : c'est la partie couvrant toutes les "section".

Activité 2.3 : Mise en place du formulaire de recrutement

Nous donnons ci-dessous les différentes informations qu'un recruteur doit renseigner dans le formulaire, ainsi que les types de champs HTML 5 correspondants.

Dans la mise en place du formulaire, il faudra regrouper les « Coordonnées de l'entreprise » et les « Propositions de l'entreprise » dans deux « fieldset » avec comme légendes (« legend ») les textes correspondants dans l'image ci-dessus.

Voici un exemple du résultat du formulaire :

Il faudra aussi, en utilisant un fichier JavaScript externalisé, faire les contrôles suivantes avant la soumission du formulaire :

  • Tout champ obligatoire (Obligatoire) doit être rempli / coché selon sa nature;

  • Le mot de passe s'il est rempli, doit être entre 8 et 10 caractères

  • Si le poste proposé est différent de « Employé », le temps de travail est automatiquement « à temps plein » (case à cocher par défaut »), sinon le choix du temps de travail est ouvert.

  • Le salaire ne peut pas être inférieur à :

    • 700 000 si le poste proposé est « Employé » et que le temps de travail est « à temp plein »

    • 500 000 si le poste proposé est « Employé » et que le temps de travail est « à mi-temp »

    • 900 000 si le poste proposé est « Collaborateur »

    • 1 000 000 si le poste proposé est « Directeur adjoint »

    • 1 500 000 si le poste proposé est « Directeur général »

//variables globales permettant de savoir si un poste / temps de travail est sélectionné
let un_poste_selectionne = false;

/* Ajouter l'événement de click au boutton permettant de soumettre le formulaire: on fait appel à la fonction
   "verifier_formulaire" permettant de vérifier si le formulaire est correctement renseigné*/
boutton_formulaire.addEventListener('click', verifier_formulaire);

//Fonction nommée auto-invoquée pour gérer les contraintes liées aux postes et aux temps de travail
(function sur_les_postes(){
    //Parcours de la liste des postes. Pour chaque poste, lui ajouter un événement click
    for(var i = 0; i < les_postes.length; i++){
        les_postes[i].onclick = function(){
            //Au clic sur un radio, on commence par mettre à "true" la variable "un_poste_selectionne"
            un_poste_selectionne = true;
            if (this.value == "employe"){ //si c'est employé qui est sélectionné, alors, ce qui suit:
                mi_temps.disabled=false;
                plein_temps.disabled=false;
                mi_temps.checked=false;
                plein_temps.checked=false;
                un_temps_de_travail_selectionne = false;
            } else {
                mi_temps.disabled=true;
                plein_temps.checked=true;
                plein_temps.disabled=true;                
                un_temps_de_travail_selectionne = true;
            }
        }
    }
    //Parcours de la liste des temps de travail. Pour chaque temps, lui ajouter un événement click
    for(var j = 0; j < les_temps.length; j++){
    les_temps[j].onclick = function(){
        //Au clic sur un checkbox (en cas de sélection seulement), on commence par mettre à "true" la variable "un_temps_de_travail_selectionne"
        if (this.checked){
            un_temps_de_travail_selectionne = true;   
        } else {
            un_temps_de_travail_selectionne = false;   
        }
             
    }
    }
})();

//fonction de vérification du formulaire
function verifier_formulaire(){
    let formulaire_ok = true;
    let message_ = "Erreurs :\n";          
    //Tests de conformité du salaire
    if ((employe.checked && mi_temps.checked && salaire.value < 500000) ||
        (employe.checked && plein_temps.checked && salaire.value < 700000) ||
       (collaborateur.checked && salaire.value < 900000)||
       (dga.checked && salaire.value < 1000000)||
       (dg.checked && salaire.value < 1500000)){
        formulaire_ok = false;
        message_ = message_ + "\n" + message_erreur_salaire;
        salaire.style.backgroundColor='red';
    }    
    //Test si un poste est sélectionné
    if (!un_poste_selectionne){
        formulaire_ok = false;
        message_ = message_ + "\n" + message_erreur_poste;
    }
    
    //Test si un temps de travail est sélectionné
    if (!un_temps_de_travail_selectionne){
        formulaire_ok = false;
        message_ = message_ + "\n" + message_erreur_temps_de_travail;
    }            
    //Marathon de Test de champ obligatoire
    if (nom_entreprise.value==''){
        formulaire_ok = false;
        message_ = message_ + "\n" + message_erreur_obligatoire;
        nom_entreprise.style.backgroundColor='red';
    }
    if (email_entreprise.value==''){
        formulaire_ok = false;
        message_ = message_ + "\n" + message_erreur_obligatoire;
        email_entreprise.style.backgroundColor='red';
    }
    if (date_demarrage.value==''){
        formulaire_ok = false;
        message_ = message_ + "\n" + message_erreur_date;
        date_demarrage.style.backgroundColor='red';
    }
    if (mot_de_passe_entreprise.value!='' && 
        (mot_de_passe_entreprise.value.length < min_mot_de_passe || mot_de_passe_entreprise.value.length > max_mot_de_passe)){        
        formulaire_ok = false;
        message_ = message_ + "\n" + message_erreur_mot_de_passe;
        mot_de_passe_entreprise.style.backgroundColor='red';
    }
    //On peut maintenant vérifier si le formulaire est Ok
    if (formulaire_ok){
        message_ = message_ok_formulaire;
        //Si tout Ok, on peut maintenant soumettre le formulaire (on attend l'année prochaine pour le traiter avec du PHP)
        recrutement_form.submit();
    } else{
        //Sinon, on reste sur la page
    }
    //Dans tous, les cas (qu'il y ait errerurs ou non), on affiche le message approprié
    alert(message_);
}
